def readiness_score(attendance_pct, diary_entries, task_count):
# Allocation: 40% Weightage
att_component = attendance_pct * 0.40
# Allocation: 30% Weightage (Scale base 15)
diary_component = (diary_entries / 15) * 100 * 0.30
# Allocation: 30% Weightage (Scale base 10)
task_component = (task_count / 10) * 100 * 0.30
raw_total = att_component + diary_component + task_component
if raw_total > 100:
return 100.0
else:
return round(raw_total, 2)
